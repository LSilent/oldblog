python生成目录树

Date:2012-10-28

code:

import os

def tree(d, n = 0):
    for i in os.listdir(d):
        p = os.path.join(d,i)
        print n*'|   '+'|__'+os.path.basename(p)
        if os.path.isdir(p):
            tree(p, n+1)

tree('./')
